Prototypes!

Big idea

If an object doesn't know, it asks its prototype

Lookup algorithm

// always at run-time
function getProperty(object, prop) {
  if(object.hasOwnProperty(prop)) {
    return object[prop];
  } else {
    if(object.__proto__) {
      getProperty(object.__proto__, prop);
    } else {
      return undefined;
    }
  }
}
